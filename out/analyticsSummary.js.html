'use strict'

var constants = require('../lib/constants.js');
var framework = require('eze-e2e-webframework');
var analyticsSummary = require('../pageObjects/Controls/analyticsSummary_control.js');
var analyticsSummarySettings = require('../pageObjects/Controls/analyticsSummary_settings_control.js');

module.exports = {

    /** 
     * Add an Analytics Summary widget to an Analytics workroom.
     * 
     * @param {int} analyticsSummaryWidgetId - the identifier of the widget.
    */
    addAnalyticsSummaryWidget: function(analyticsSummaryWidgetId) {
        framework.utils.dismissAllNotificationsIfAny();
        framework.dashboard.addWidget(constants.AnalyticsProductName, constants.AnalyticsSummary);
        analyticsSummary.checkIfSummaryLoaded(analyticsSummaryWidgetId);
    },

    /** 
     * Change title with a given input string.
     * 
     * @param {int} analyticsSummaryWidgetId - the identifier of the widget.
     * @param {string} newTitle - new title name to replace existing title.
     */
    changeTitle: function(analyticsSummaryWidgetId, newTitle) {
        analyticsSummarySettings.changeTitleField(newTitle);
    },

     /** 
     * Change portfolio field with a given array of inputs.
     * 
     * @param {int} analyticsSummaryWidgetId - the identifier of the widget.
     * @param {string[]} portInput - Array of Portfolios to filter on.
     */
    changeWidgetSettingsPortfolio: function(analyticsSummaryWidgetId, portfolioInput) {
        analyticsSummarySettings.changeWidgetSettingsPortfolioField(portfolioInput);
    },

    /** 
     * Change widget settings view configuration with a given input.
     * 
     * @param {int} analyticsSummaryWidgetId - the identifier of the widget.
     * @param {string} viewConfigurationInput - view configuration to select.
     */
    changeViewConfiguration: function(analyticsSummaryWidgetId, viewConfigurationInput) {
        analyticsSummarySettings.changeViewConfigurationField(viewConfigurationInput);
    },

    /** 
     * Change included position states field with a given position state.
     * 
     * @param {int} analyticsSummaryWidgetId - the identifier of the widget.
     * @param {string} positionStateInput - position state input to select.

        analyticsSummarySettings.changeIncludedPositionStatesField(positionStateInput);

    /** 
     * Change flexible basis column field with a given input column name.
     * 
     * @param {int} analyticsSummaryWidgetId - the identifier of the widget.

        analyticsSummarySettings.changeFlexibleBasisColumnField(flexibleBasisInput);

    /** 
     * Clicks "Include Closed Positions" checkbox based on desired selection value.

     * @param {int} analyticsSummaryWidgetId - the identifier of the widget.
     * @param {boolean} includeClosedPositions - whether or not to have checkbox checked/unchecked.

     * @param {int} analyticsSummaryWidgetId - the identifier of the widget.

    /** 
     * Verify from analytics summary widget.
     * 
     * @param {int} summaryWidgetId - the identifier of the widget.

        analyticsSummary.verifyAnalyticsSummaryIsNotBlank(summaryWidgetId);
    },

  /** 
     * Check that a best worst has a certain portfolio filter
     * 
     * @param {int} summaryWidgetId - the identifier of the widget.
     * @param {string} selectedPort - portfolio you expected to be filtered on.
     */
    verifyPortfolioFilter: function(summaryWidgetId, selectedPort) {
        var dropdown = element.all(By.id("multiSelectPortfolio_taglist")).last();
        var port = dropdown.element(By.css("span[unselectable = 'on']"));
        port.getText()
        .then((text)=>{
            console.log("The Portfolio filter is " + text);
            expect(text).toBe(selectedPort);   
        });  
    },

    /** 
     * Check that summary widget settings are the correct configuration based on input settings
     * 
     * @param {object} widgetSettings - object of widget settings that should be configured against.
     */
    verifySummaryWidgetSettings: function(widgetSettings) {
        analyticsSummarySettings.verifySummaryWidgetSettings(widgetSettings);
    }

}
