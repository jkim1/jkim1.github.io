     */
     static getAppKey() {
        page.Info();
        something.client();
        var licenseKey = shellPage.getLicenseKey().then(keyObject => {
            var firmCode = browser.params.credentials.firm;
            expect(firmCode).toBe(keyObject.firmCode);

        });
        return licenseKey;
     }

     * Search in global search box with given search text

     * @returns {ElementArrayFinder} Search results
     */

        return shellPage.search(searchText);
     }

     /**
     * Rename a user page by right click and select 'Rename' option from the dropdown.
     *

     */

        userpagePage.renameUserPage(pageName, newPageName);
        var element = this.getPageSelector(newPageName);
        expect(element.isPresent()).toBeTruthy();
