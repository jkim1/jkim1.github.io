'use strict'

var framework = require('eze-e2e-webframework');
var userPages = require('eze-e2e-webframework/scripts/userPages.js')
var pageConfig = require('eze-e2e-webframework/scripts/pageConfig.js')
var logger = framework.log("Analytics");
var constants = require('../lib/constants.js');
var workroom = require('../pageObjects/workroom_control.js');

module.exports = {
    
    /**
     * Login and navigate to Analytics page.
     */
    loginToAnalytics: function() {
        browser.ignoreSynchronization = true;
        logger.debug('In beforeAll');
        framework.login.getToLoggedInState().then(() => {
            browser.ignoreSynchronization = false;
            framework.shell.navigateToProduct("Analytics");
            logger.debug('beforeAll Done');
        });
    },

    /** 
     * Login and navigate to Analytics page and then a corresponding page by name.
     * 
     * @param {string} pageName - Name of created page.
    */
    loginToAnalyticsPageByName: function(pageName) {
        browser.ignoreSynchronization = true;
        logger.debug('In beforeAll');
        framework.login.getToLoggedInState().then(() => {
            browser.ignoreSynchronization = false;
            framework.shell.navigateToProductThenPage("Analytics", pageName);
            logger.debug('beforeAll Done');
        });
    },

    /**
     * Navigate to analytics page after you have been logged in and done other things
     */
    navigateToAnalytics: function(){
        framework.shell.navigateToProduct("Analytics");
    },

    /** 
     * Create a new Analytics workroom with a provided page name.
     * 
     * @param {string} pageName - Name of created page.
    */
    createNewAnalyticsWorkroom: function(pageName) {
        framework.utils.dismissAllNotificationsIfAny();
        workroom.deleteResidualWorkroomByName(pageName);
        framework.utils.dismissAllNotificationsIfAny();
        userPages.addNewUserPage(pageName);
    },

    /** 
     * Delete an existing Analytics workroom by a provided page name.
     * 
     * @param {string} Name of existing page to delete.
    */
    deleteAnalyticsWorkroom: function(existingPageName) {
        userPages.deleteUserPage(existingPageName);
    },

    /** 
     * Add a new page configuration for the current view in a page.
     * 
     * @param {string} pageConfigName - name of page configuration to create.
    */
    addPageConfiguration: function(pageConfigName) {
        workroom.addPageConfiguration(pageConfigName);
    },

    /** 
     * Save an existing page configuration for the current view in a page.
     * 
     * @param {string} pageConfigName - name of page configuration to save.
    */
    savePageConfiguration: function(pageConfigName) {
        workroom.savePageConfiguration(pageConfigName);
    },
    
    /** 
     * Switch to an existing page configuration.
     * 
     * @param {string} pageConfigName - name of page configuration to switch to.
    */
    switchPageConfiguration: function(pageConfigName) {
        workroom.switchPageConfiguration(pageConfigName);
    }





}
