'use strict';
const webfw = require("eze-e2e-webframework");
var navAndPLPage = require("../pageObjects/NAVandPLPage");
var controls = require('../pageObjects/Controls/reportControls');
var logger = new webfw.log("Reports Script");
var accountingUtils = require("../utils");
var shell = require('./accountingShell');
const integration = require("eze-integration-protractor-framework");

/**
 * APIs for interacting with Open Lots report
 */
class NAVandPLReport {

    static runDefaultNAVandPLReport() {
        shell.navigateToReports();
        controls.getReportPageStatus().then((pageStatus) => {
            navAndPLPage.selectNavAndPLReport(pageStatus);
                return controls.clickApply(pageStatus).then(() => {
                    logger.debug('Clicked  Apply button to run the report');
                    console.log('Waiting for report to load');
                    return webfw.utils.waitForElementNotPresent(controls.elements.reportContentBusyIndicator, 'Busy Indicator').then(() => {
                        console.log('Report loaded');

                });
            });
        });
    }

    /**
     * Run a NAV and P&L accounting report based on Start Business Date and End Business Date.
     *
     * Selects NAV and PL report from the dropdown.
     * @param {dateFormat} dateValue is a date for Business date field.
     * @param {dateFormat} dateValue is a date for Business date field.
     *
     * Note: would like to add the ability to specify different parameters to this function eventually.
     */
    static runNAVandPLReport(fromDateValue, toDateValue) {
        shell.navigateToReports();
        controls.getReportPageStatus().then((pageStatus) => {
            navAndPLPage.selectNavAndPLReport(pageStatus);
            return navAndPLPage.setStartBusinessDate(pageStatus, fromDateValue).then(() => {
                console.log('set start business date');
                navAndPLPage.verifyStartBusinessDate(pageStatus, fromDateValue);
                return navAndPLPage.setEndBusinessDate(pageStatus, toDateValue).then(() => {
                    console.log('set end business date');
                    navAndPLPage.verifyEndBusinessDate(pageStatus, toDateValue);
                        console.log('report settings changed');
                        return controls.clickApply(pageStatus).then(() => {
                            logger.debug('Clicked  Apply button to run the report');
                            console.log('Waiting for report to load');
                            return webfw.utils.waitForElementNotPresent(controls.elements.reportContentBusyIndicator, 'Busy Indicator').then(() => {
                                console.log('Report loaded');
                        });
                    });
                });
            });
        });
    }

    static verifyHistPLNotZero() {
        shell.navigateToReports();
        browser.actions().doubleClick(controls.elements.histPLColHeader).perform();
        return controls.getTextQuantity(controls.elements.firstHistPLCell).then((resultValue) => {
          console.log('HistPL value is ' + resultValue);
          expect(resultValue).not.toBe('0.00');
          expect(resultValue).not.toBe('');
        });
    }

}
module.exports = NAVandPLReport;
